16  package com.okta.spring.example.controllers;
17  
18  import com.okta.idx.sdk.api.client.IDXAuthenticationWrapper;
19  import com.okta.idx.sdk.api.exception.ProcessingException;
20  import com.okta.idx.sdk.api.model.IDXClientContext;
21  import com.okta.idx.sdk.api.response.ErrorResponse;
22  import com.okta.spring.boot.oauth.config.OktaOAuth2Properties;
23  import org.springframework.beans.factory.annotation.Autowired;
24  import org.springframework.stereotype.Controller;
25  import org.springframework.web.bind.annotation.GetMapping;
26  import org.springframework.web.bind.annotation.RequestParam;
27  import org.springframework.web.servlet.ModelAndView;
28  
29  import javax.servlet.http.HttpServletRequest;
30  import javax.servlet.http.HttpSession;
31  import java.net.MalformedURLException;
32  import java.net.URL;
33  import java.security.NoSuchAlgorithmException;
34  
35  @Controller
36  public class LoginController {
37  
38      private static final String STATE = "state";
39      private static final String NONCE = "nonce";
40      private static final String SCOPES = "scopes";
41      private static final String OKTA_BASE_URL = "oktaBaseUrl";
42      private static final String OKTA_CLIENT_ID = "oktaClientId";
43      private static final String REDIRECT_URI = "redirectUri";
44      private static final String ISSUER_URI = "issuerUri";
45      /* idx related properties */
46      private static final String INTERACTION_HANDLE = "interactionHandle";
47      private static final String CODE_VERIFIER = "codeVerifier";
48      private static final String CODE_CHALLENGE = "codeChallenge";
49      private static final String CODE_CHALLENGE_METHOD = "codeChallengeMethod";
50      private static final String CODE_CHALLENGE_METHOD_VALUE = "S256";
51      private static final String IDX_CLIENT_CONTEXT = "idxClientContext";
52  
53      private final OktaOAuth2Properties oktaOAuth2Properties;
54  
55      @Autowired
56      private IDXAuthenticationWrapper idxAuthenticationWrapper;
57  
58      private IDXClientContext idxClientContext;
59  
60      public LoginController(OktaOAuth2Properties oktaOAuth2Properties) {
61          this.oktaOAuth2Properties = oktaOAuth2Properties;
62      }
63  
64      @GetMapping(value = "/custom-login")
65      public ModelAndView login(HttpServletRequest request,
66                                @RequestParam(name = "state", required = false) String state,
67                                @RequestParam(name = "nonce") String nonce,
68                                HttpSession session) throws MalformedURLException, NoSuchAlgorithmException {
69  
70          if (session.getAttribute(IDX_CLIENT_CONTEXT) == null) {
71              try {
72                  idxClientContext = idxAuthenticationWrapper.getClientContext();
73              } catch (ProcessingException e) {
74                  ModelAndView modelAndView = new ModelAndView("error");
75                  ErrorResponse errorResponse = e.getErrorResponse();
76                  if (errorResponse != null) {
77                      modelAndView.addObject("errorDetails",
78                              errorResponse.getError() + "," + errorResponse.getErrorDescription());
79                  } else {
80                      modelAndView.addObject("errorDetails", "Unknown error");
81                  }
82                  return modelAndView;
83              }
84              session.setAttribute(IDX_CLIENT_CONTEXT, idxClientContext);
85          }
86  
87          if (idxClientContext == null) {
88              ModelAndView modelAndView = new ModelAndView("error");
89              modelAndView.addObject("error_details", "Unknown error");
90              return modelAndView;
91          }
92  
93          // if we don't have the state parameter redirect
94          if (state == null) {
95              return new ModelAndView("redirect:" + oktaOAuth2Properties.getRedirectUri());
96          }
97  
98          String issuer = oktaOAuth2Properties.getIssuer();
99          // the widget needs the base url, just grab the root of the issuer
100         String orgUrl = new URL(new URL(issuer), "/").toString();
101 
102         ModelAndView mav = new ModelAndView("login");
103         mav.addObject(STATE, state);
104         mav.addObject(NONCE, nonce);
105         mav.addObject(SCOPES, oktaOAuth2Properties.getScopes());
106         mav.addObject(OKTA_BASE_URL, orgUrl);
107         mav.addObject(OKTA_CLIENT_ID, oktaOAuth2Properties.getClientId());
108         mav.addObject(INTERACTION_HANDLE, idxClientContext.getInteractionHandle());
109         mav.addObject(CODE_VERIFIER, idxClientContext.getCodeVerifier());
110         mav.addObject(CODE_CHALLENGE, idxClientContext.getCodeChallenge());
111         mav.addObject(CODE_CHALLENGE_METHOD, CODE_CHALLENGE_METHOD_VALUE);
112 
113         // from ClientRegistration.redirectUriTemplate, if the template is change you must update this
114         mav.addObject(REDIRECT_URI,
115             request.getScheme() + "://" + request.getServerName() + ":" + request.getServerPort() +
116             request.getContextPath() + "/authorization-code/callback"
117         );
118         mav.addObject(ISSUER_URI, issuer);
119 
120         session.setAttribute(CODE_VERIFIER, idxClientContext.getCodeVerifier());
121         return mav;
122     }
123 
124     @GetMapping("/post-logout")
125     public String logout(HttpSession session) {
126         session.invalidate();
127         return "logout";
128     }
129 
130     @GetMapping("/403")
131     public String error403() {
132         return "403";
133     }
134 }
